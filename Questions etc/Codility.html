7.

  var moo: String { get }
}

  var moo: String {
    return "Moo"
  }
  func test() {
     print("Foo \(moo)")
  }
}
 
class Bar: Foo {
  var moo: String {
    return "Mooo"
  }
  func test() {
    print("Bar \(moo)")
  }
}
 
let b: Foo = Bar()
b.test()

B: Bar Mooo
C: Foo Moo
D: Foo Mooo
